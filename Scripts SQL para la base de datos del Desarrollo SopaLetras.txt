-- creación procedimientos necesarios

-- procedimiento que realiza la actualizacion de la cuenta de los resultados

USE [SopaLetras]
GO

SET ANSI_NULLS ON
GO

SET QUOTED_IDENTIFIER ON
GO

CREATE PROCEDURE [dbo].[cuenta_resultado]
	@resultado Bit
AS
BEGIN

	DECLARE @TotalFilas INT;

	SELECT @TotalFilas = COUNT(*) FROM Cuenta;

	IF @TotalFilas = 0
	BEGIN
		INSERT INTO Cuenta (contieneNombre, noContieneNombre) VALUES(NULL, NULL);
	END

	IF @resultado = 1
	BEGIN

		update Cuenta
		set contieneNombre = 1 + (SELECT 
			CASE 
				WHEN NOT EXISTS (SELECT 1 FROM Cuenta) THEN 0 -- La tabla no está vacía, retorna 0
				ELSE (SELECT ISNULL(c.contieneNombre, 0) FROM Cuenta c) -- La tabla está vacía, retorna 1
			END AS Resultado
		);
	END
	ELSE
	BEGIN
		update Cuenta
		set noContieneNombre = 1 + (SELECT 
			CASE 
				WHEN NOT EXISTS (SELECT 1 FROM Cuenta) THEN 0 -- La tabla no está vacía, retorna 0
				ELSE (SELECT ISNULL(c.noContieneNombre, 0) FROM Cuenta c) -- La tabla está vacía, retorna 1
			END AS Resultado
		);
	END

END
GO


-- Procedimiento que se usa para el reporte de los datos guardados y su relación

USE [SopaLetras]
GO

SET ANSI_NULLS ON
GO

SET QUOTED_IDENTIFIER ON
GO

CREATE PROCEDURE [dbo].[reporte_cuenta]
AS
BEGIN
	SELECT ISNULL(c.contieneNombre, 0) contieneNombre, ISNULL(c.noContieneNombre, 0) noContieneNombre, ISNULL((c.contieneNombre / c.noContieneNombre), 0) relacion from Cuenta c;
END
GO
